val with_timeout : float ->
       f:(unit -> ('a, [> `io_exn of exn | `timeout of float ] as 'b) Flow_base.t) ->
       ('a, 'b) Flow_base.t
Execute a function f with a timeout (in seconds). If f throws an exception it will be passed as `io_exn e, if the functions timeouts the error will be `timeout time.

val file_info : ?follow_symlink:bool ->
       string ->
       ([ `absent
        | `block_device
        | `character_device
        | `directory
        | `fifo
        | `file of int
        | `socket
        | `symlink of string ],
        [> `system of [> `file_info of string ] * [> `exn of exn ] ])
       Flow_base.t
Get information about a path (whether it exists, its size, or sym-link destination). If follow_symlink is false (default) use lstat (so the result can be `symlink _), if true call stat (information about the target).
